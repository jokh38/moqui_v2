139 CUDA_DEVICE
140 inline virtual R
141 stopping_power_ratio(R Ek, int8_t id = -1) {
142 // Convert density from g/mm^3 to g/cm^3 for the formula
143 R density_tmp = this->rho_mass * 1000.0;
144
145 if (density_tmp > 0.26) {
146 // Empirical formula for materials denser than 0.26 g/cm^3
147 R rsp = 1.0123 - 3.386e-5 * Ek;
148 rsp += 0.291 * (1.0 + mqi::mqi_pow(Ek, static_cast<R>(-0.3421))) *
149 (mqi::mqi_pow(density_tmp, static_cast<R>(-0.7)) - 1.0);
150
151 if (density_tmp > 0.9)
152 return 1.0; // For high-density materials, approximate as 1.0
153 else
154 // For intermediate densities, interpolate between a known point and the calculated rsp.
155 return mqi::intpl1d<R>(density_tmp, 0.26, 0.9, 0.9925, rsp);
156 } else {
157 // For low-density materials
158 if (density_tmp < 0.0012) {
159 return 0.0; // For near-vacuum, ratio is zero.
160 } else {
161 // Interpolate for other low-density materials like lung.
162 return mqi::intpl1d<R>(density_tmp, 0.0012, 0.26, 0.8815, 0.9925);
163 }
164 }
165 }

175 CUDA_HOST_DEVICE
176 virtual R
177 radiation_length() {
178 R radiation_length_mat = 0.0;
179 R f = 0.0;
180 R density_tmp = this->rho_mass * 1000.0; // Convert to kg/m^3 for formula
181 // Fippel's empirical formula based on density regimes
182 if (density_tmp <= 0.26) {
183 f = 0.9857 + 0.0085 * density_tmp;
184 } else if (density_tmp > 0.26 && density_tmp <= 0.9) {
185 f = 1.0446 - 0.2180 * density_tmp;
186 } else if (density_tmp > 0.9) {
187 f = 1.19 + 0.44 * logf(density_tmp - 0.44);
188 }
189 radiation_length_mat = (this->units.water_density * this->units.radiation_length_water) /
190 (density_tmp * 1e-3 * f);
191 return radiation_length_mat;
192 }
