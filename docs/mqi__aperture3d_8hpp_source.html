235 mqi::vec2<R> pos0 = segment[0];
236 mqi::vec2<R> pos1;
237 float min_y, max_y, max_x, intersect_x;
238 int count = 0;
239 int i, j, c = 0;
240 for (i = 0, j = num_segment - 1; i < num_segment; j = i++) {
241 pos0 = segment[i];
242 pos1 = segment[j];
243 if ((((pos0.y <= pos.y) && (pos.y < pos1.y)) ||
244 ((pos1.y <= pos.y) && (pos.y < pos0.y))) &&
245 (pos.x < (pos1.x - pos0.x) * (pos.y - pos0.y) / (pos1.y - pos0.y) + pos0.x)) {
246 c = !c;
247 }
248 }

279 intersect(mqi::vec3<R>& p, mqi::vec3<R>& d, mqi::vec3<ijk_t> idx) {
280 mqi::intersect_t<R> its;
281 its.cell = idx;
282 its.side = mqi::NONE_XYZ_PLANE;
283 mqi::intersect_t<R> its_out;
284 mqi::intersect_t<R> its_in;
285 its_out.dist = 0;
286 its_out.cell.x = -5;
287 its_out.cell.y = -5;
288 its_out.cell.z = -5;
289 its_in.dist = 0;
290 its_in.cell.x = -10;
291 its_in.cell.y = -10;
292 its_in.cell.z = -10;
293
294 if (!is_inside(p)) {
295 its_out.type = mqi::APERTURE_CLOSE;
296 return its_out;
297 } else {
298 if (d.z < 0) {
299 if (mqi::mqi_abs(p.z - this->ze_[0]) < 1e-3) {
300 its_in.dist = 0.0;
301 } else {
302 its_in.dist = -(p.z - this->ze_[0]) / d.z;
303 }
304
305 } else if (d.z > 0) {
306 if (mqi::mqi_abs(p.z - this->ze_[1]) < 1e-3) {
307 its_in.dist = 0;
308 } else {
309 its_in.dist = (this->ze_[1] - p.z) / d.z;
310 }
311 } else {
312 its_in.dist = 0;
313 }
314 assert(its_in.dist >= 0);
315 its_in.type = mqi::APERTURE_OPEN;
316 return its_in;
317 }
318 }
