92 ct(std::string f, bool is_print = false) {
93 // Manually allocate and copy the directory path string.
94 ct_dir = new char[f.length() + 1];
95 strcpy(ct_dir, f.c_str());
96
97 // Use GDCM to load all files from the specified directory (non-recursively).
98 gdcm::Directory dir;
99 dir.Load(this->ct_dir, false);
100 const gdcm::Directory::FilenamesType& all_files = dir.GetFilenames();
101
102 // Scan files to find those with Modality (0008,0060) set to "CT".
103 gdcm::Scanner scanner;
104 const gdcm::Tag ct_tag = gdcm::Tag(0x08, 0x60);
105 scanner.AddTag(ct_tag);
106 scanner.Scan(all_files);
107 auto ct_files = scanner.GetAllFilenamesFromTagToValue(ct_tag, "CT");
108
109 // Sort the CT files by their spatial position to ensure correct slice order.
110 gdcm::IPPSorter ippsorter;
111 ippsorter.SetComputeZSpacing(false);
112 ippsorter.Sort(ct_files);
113 files_ = ippsorter.GetFilenames();
114
115 // Scan the sorted files to extract necessary metadata tags.
116 gdcm::Scanner s;
117 s.AddTag(gdcm::Tag(0x0008, 0x0018));
118 s.AddTag(gdcm::Tag(0x0020, 0x0032));
119 s.AddTag(gdcm::Tag(0x0028, 0x0010));
120 s.AddTag(gdcm::Tag(0x0028, 0x0011));
121 s.AddTag(gdcm::Tag(0x0028, 0x0030));
122 s.AddTag(gdcm::Tag(0x0018, 0x0050));
123 if (!s.Scan(files_)) assert("scan fail.");
124
125 size_t nx;
126 size_t ny;
127 size_t nz = files_.size();
128
129 // Initialize grid properties and manually allocate coordinate arrays.
130 this->z_ = new R[nz];
131 this->dim_.z = nz;
132 double x0, y0;
133 dz_ = new R[nz];
134 float dz;
135 for (size_t i = 0; i < nz; ++i) {
136 gdcm::Scanner::TagToValue const& m0 = s.GetMapping(files_[i].c_str());
137
138 // Parse the Image Position (Patient) tag, which is a string like "x\y\z".
139 std::string img_position(m0.find(gdcm::Tag(0x0020, 0x0032))->second);
140 unsigned int deli0 = img_position.find_first_of('\\');
141 unsigned int deli1 = img_position.find_last_of('\\');
142 this->z_[i] = (R) (std::stod(img_position.substr(deli1 + 1)));
143 dz_[i] = std::stod(m0.find(gdcm::Tag(0x0018, 0x0050))->second);
144
145 // Determine grid parameters from the first image slice.
146 if (i == 0) {
147 x0 = std::stod(img_position.substr(0, deli0));
148 y0 = std::stod(img_position.substr(deli0 + 1, deli1 - deli0));
149
150 ny = std::stoi(m0.find(gdcm::Tag(0x0028, 0x0010))->second); // Rows
151 nx = std::stoi(m0.find(gdcm::Tag(0x0028, 0x0011))->second); // Columns
152 this->dim_.x = nx;
153 this->dim_.y = ny;
154 // Parse Pixel Spacing tag, which is a string like "dx\dy".
155 std::string pixel_spacing(m0.find(gdcm::Tag(0x0028, 0x0030))->second);
156 unsigned int deli = pixel_spacing.find_first_of('\\');
157 dx_ = std::stod(pixel_spacing.substr(0, deli));
158 dy_ = std::stod(pixel_spacing.substr(deli + 1));
159 }
160
161 // Map SOP Instance UID to file path for quick lookup.
162 uid2file_.insert(std::make_pair(m0.find(gdcm::Tag(0x0008, 0x0018))->second, files_[i]));
163 }
164
165 // Populate the x and y coordinate arrays.
166 this->x_ = new R[nx];
167 for (size_t i = 0; i < nx; ++i) {
168 this->x_[i] = x0 + dx_ * i;
169 }
170 this->y_ = new R[ny];
171 for (size_t i = 0; i < ny; ++i) {
172 this->y_[i] = y0 + dy_ * i;
173 }
174
175 // Print summary if requested.
176 if (is_print) {
177 std::cout << "Reading DICOM directory.. : Getting patient CT info.." << std::endl;
178 std::cout << "Reading DICOM directory.. : Detected CT info is as below." << std::endl;
179 std::cout << "CT (nx, ny, nz) -> (" << this->dim_.x << ", " << this->dim_.y << ", "
180 << this->dim_.z << ")" << std::endl;
181 std::cout << "CT (dx, dy, dz) -> (" << dx_ << ", " << dy_ << ", "
182 << this->z_[1] - this->z_[0] << ")" << std::endl;
183 std::cout << "CT (x, y, z) -> (" << this->x_[0] << ", " << this->y_[0] << ", "
184 << this->z_[0] << ")" << std::endl;
185 }
186 }

207 virtual void
208 load_data() {
209 std::cout << "Reading DICOM directory.. : Loading patient CT pixel data.." << std::endl;
210 size_t nb_voxels_2d = this->dim_.x * this->dim_.y;
211 size_t nb_voxels_3d = nb_voxels_2d * this->dim_.z;
212
213 this->data_.resize(nb_voxels_3d);
214 float intercept = 0;
215 float slope = 1;
216
217 for (size_t i = 0; i < this->dim_.z; ++i) {
218 gdcm::ImageReader reader;
219 reader.SetFileName(files_[i].c_str());
220 reader.Read();
221 const gdcm::Image& img = reader.GetImage();
222
223 // Extract slope and intercept for HU conversion.
224 intercept = float(img.GetIntercept());
225 slope = float(img.GetSlope());
226
227 // Read the raw pixel data into the correct slice of the 3D grid.
228 gdcm::PixelFormat pixeltype = img.GetPixelFormat();
229 if (pixeltype == gdcm::PixelFormat::INT16) {
230 img.GetBuffer((char*) &this->data_[i * nb_voxels_2d]);
231 } else {
232 // Production code should handle other pixel types if they are expected.
233 assert(0);
234 }
235 }
236 // Apply rescale slope and intercept to the entire dataset to get Hounsfield Units.
237 this->data_ = this->data_ * int16_t(slope) + intercept;
238 std::cout << "Reading DICOM directory.. : Patient CT pixel data successfully loaded." << std::endl;
239 }
