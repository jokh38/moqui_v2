74 uint8_t*
75 read_mha_file(std::string filename) {
76 std::string line;
77 std::ifstream fid(filename, std::ios::ate | std::ios::binary); // Open in binary mode
78 if (!fid.is_open()) {
79 throw std::runtime_error("Cannot open mask file: " + filename);
80 }
81 fid.seekg(0);
82 std::string delimeter = "=";
83 size_t pos, pos_current, pos_prev, total_size;
84 std::string parameter, value;
85 uint16_t nx, ny, nz;
86 uint8_t* mask = nullptr;
87 // Parse the MHA header
88 while (std::getline(fid, line)) {
89 line = trim_copy(line);
90 pos = line.find(delimeter);
91 parameter = trim_copy(line.substr(0, pos));
92 if (strcasecmp(parameter.c_str(), "DimSize") == 0) {
93 value = trim_copy(line.substr(pos + 1));
94 pos_prev = 0;
95 pos_current = value.find(" ");
96 nx = std::atoi(value.substr(pos_prev, pos_current).c_str());
97 pos_prev = pos_current + 1;
98 pos_current = value.find(" ", pos_prev);
99 ny = std::atoi(value.substr(pos_prev, pos_current).c_str());
100 pos_prev = pos_current + 1;
101 pos_current = value.find(" ", pos_prev);
102 nz = std::atoi(value.substr(pos_prev, pos_current).c_str());
103 } else if (strcasecmp(parameter.c_str(), "ElementDataFile") == 0) {
104 value = trim_copy(line.substr(pos + 1));
105 if (strcasecmp(value.c_str(), "LOCAL") != 0) {
106 throw std::runtime_error("Mask files does not contain data.");
107 }
108 break; // Stop parsing after finding the data file entry
109 }
110 }
111 total_size = nx * ny * nz;
112 mask = new uint8_t[total_size];
113 // Read the binary data block
114 fid.read((char*) (mask), total_size * sizeof(uint8_t));
115 fid.close();
116 return mask;
117 }

125 CUDA_HOST
126 void
127 read_mask_files() {
128 if (mask_filenames.empty()) {
129 throw std::runtime_error("Mask filelist are required for masking scorers.");
130 }
131 // Allocate and initialize the total mask buffer.
132 this->mask_total = new uint8_t[this->ct_size];
133 std::memset(this->mask_total, 0, this->ct_size * sizeof(uint8_t));
134
135 uint8_t* mask_temp = nullptr;
136 for (const auto& filename : mask_filenames) {
137 printf("Reading maskfile %s\n", filename.c_str());
138 try {
139 mask_temp = read_mha_file(filename);
140 for (size_t i = 0; i < this->ct_size; i++) {
141 assert(mask_temp[i] == 0 || mask_temp[i] == 1);
142 this->mask_total[i] |= mask_temp[i]; // Use logical OR to combine masks
143 }
144 delete[] mask_temp; // Free the temporary mask buffer
145 mask_temp = nullptr;
146 } catch (...) {
147 if (mask_temp) delete[] mask_temp;
148 throw; // Re-throw the exception
149 }
150 }
151 }

250 CUDA_HOST
251 mqi::roi_t*
252 mask_to_roi() {
253 uint32_t original_size = this->ct_size;
254 uint32_t length = 0;
255 mqi::roi_mapping_t method = mqi::CONTOUR;
256 std::vector<uint32_t> start_vec;
257 std::vector<uint32_t> stride_vec;
258 std::vector<uint32_t> acc_stride_vec;
259 bool ind_started = false;
260 int32_t cnb, start_ind, acc_stride_tmp;
261 for (ijk_t z_ind = 0; z_ind < this->ct_dim.z; z_ind++) {
262 for (ijk_t y_ind = 0; y_ind < this->ct_dim.y; y_ind++) {
263 for (ijk_t x_ind = 0; x_ind < this->ct_dim.x; x_ind++) {
264 if (this->mask_total[ijk2cnb(x_ind, y_ind, z_ind)] == 1 && !ind_started) {
265 ind_started = true;
266 start_vec.push_back(ijk2cnb(x_ind, y_ind, z_ind));
267 }
268 if (this->mask_total[ijk2cnb(x_ind, y_ind, z_ind)] == 0 && ind_started) {
269 ind_started = false;
270 if (acc_stride_vec.size() > 0)
271 acc_stride_tmp = acc_stride_vec.back();
272 else
273 acc_stride_tmp = 0;
274 start_ind = start_vec.back();
275 stride_vec.push_back(ijk2cnb(x_ind, y_ind, z_ind) - start_ind);
276 acc_stride_vec.push_back(acc_stride_tmp + stride_vec.back());
277 }
278 }
279 }
280 }
281 uint32_t* start = new uint32_t[start_vec.size()];
282 uint32_t* stride = new uint32_t[start_vec.size()];
283 uint32_t* acc_stride = new uint32_t[acc_stride_vec.size()];
284
285 std::copy(start_vec.begin(), start_vec.end(), start);
286 std::copy(stride_vec.begin(), stride_vec.end(), stride);
287 std::copy(acc_stride_vec.begin(), acc_stride_vec.end(), acc_stride);
288 length = start_vec.size();
289 mqi::roi_t* roi = new mqi::roi_t(method, original_size, length, start, stride, acc_stride);
290 return roi;
291 }

327 std::vector<std::string>
328 read_input_parameters() {
329 std::ifstream fid(this->filename);
330 std::string line;
331 std::vector<std::string> parameters_total;
332 size_t comment;
333 if (fid.is_open()) {
334 while (getline(fid, line)) {
335 if (line.empty()) continue;
336 line = trim_copy(line);
337 if (line.at(0) == '#') continue; // Skip comment lines
338 comment = line.find("#");
339 if (comment != std::string::npos) {
340 line = line.substr(0, comment);
341 } // Ignore inline comments
342 parameters_total.push_back(line);
343 }
344 fid.close();
345 } else {
346 throw std::runtime_error("Cannot open input parameter file.");
347 }

403 size_t pos_current = 0, pos_prev = 0;
404 std::string value = get_string(option, "");
405 std::vector<std::string> values;
406 if (value.empty()) return values;
407 pos_current = value.find(delimeter1);
408 std::string temp;
409 // Loop to find all occurrences of the delimiter

411 temp = trim_copy(value.substr(pos_prev, pos_current - pos_prev));
412 if (!temp.empty()) {
413 values.push_back(temp);

415 pos_prev = pos_current + 1;
416 pos_current = value.find(delimeter1, pos_prev);
417 }
418 // Add the last token after the final delimiter
419 temp = trim_copy(value.substr(pos_prev, pos_current - pos_prev));
420 if (!temp.empty()) {
421 values.push_back(temp);
