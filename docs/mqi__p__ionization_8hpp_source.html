309 CUDA_HOST_DEVICE
310 virtual R
311 cross_section(const relativistic_quantities<R>& rel, const material_t<R>& mat) {
312 R cs = 0;
313 if (rel.Ek >= Ei && rel.Ek <= Ef) {
314 // Find table indices for linear interpolation
315 uint16_t idx0 = uint16_t((rel.Ek - Ei) / this->E_step);
316 uint16_t idx1 = idx0 + 1;
317 R x0 = this->Ei + idx0 * this->E_step;
318 R x1 = x0 + this->E_step;
319 // Linearly interpolate to find the cross-section at the particle's energy
320 cs = mqi::intpl1d<R>(rel.Ek, x0, x1, cs_table[idx0], cs_table[idx1]);
321 }
322 // Convert from mm^2/g to cm^2/g, then multiply by density (g/cm^3) to get cm^-1
323 cs *= 0.01 * mat.rho_mass;
324 return cs;
325 }
326
334 CUDA_HOST_DEVICE
335 virtual inline R
336 dEdx(const relativistic_quantities<R>& rel, const material_t<R>& mat) {
337 R pw = 0;
338 if (rel.Ek >= Ei && rel.Ek <= Ef) {
339 uint16_t idx0 = uint16_t((rel.Ek - Ei) / this->E_step);
340 uint16_t idx1 = idx0 + 1;
341 R x0 = Ei + idx0 * this->E_step;
342 R x1 = x0 + this->E_step;
343 pw = mqi::intpl1d<R>(rel.Ek, x0, x1, pw_table[idx0], pw_table[idx1]);
344 } else if (rel.Ek < Ei && rel.Ek > 0) {
345 pw = pw_table[0]; // For energies below the table, use the first value.
346 assert(pw >= 0);
347 }
348 // Return dE/dx in MeV/cm (pw is in MeV*cm^2/g, rho_mass is in g/cm^3)
349 return -1.0 * pw * mat.rho_mass;
350 }

358 CUDA_HOST_DEVICE
359 inline R
360 sample_delta_energy(const R Te_max, const mqi_rng* rng) {
361 R eta = mqi_uniform<R>(rng);
362 // Inverse transform sampling from the 1/T^2 differential cross-section.
363 return Te_max * this->T_cut / ((1.0 - eta) * Te_max + eta * this->T_cut);
364 }

378 CUDA_HOST_DEVICE
379 virtual inline R
380 energy_loss(const relativistic_quantities<R>& rel,
381 material_t<R>& mat,
382 const R step_length,
383 mqi_rng* rng) {
384 // Convert physical step length to water-equivalent path length
385 R length_in_water = step_length * mat.stopping_power_ratio(rel.Ek) * mat.rho_mass /
386 this->units.water_density;
387
388 // Find current range in water from the table
389 uint16_t n = uint16_t((rel.Ek - this->Ei) / this->E_step);
390 R x0 = this->Ei + n * this->E_step;
391 R x1 = x0 + this->E_step;
392 if (x0 > rel.Ek) n -= 1;
393 if (x1 < rel.Ek) n += 1;
394 R r = mqi::intpl1d(rel.Ek, x0, x1, r_steps[n], r_steps[n + 1]);
395
396 if (r < length_in_water) return rel.Ek; // If step is longer than range, particle stops.
397 r -= length_in_water; // Calculate residual range.
398
399 // Find the energy corresponding to the residual range
400 do {
401 if (r >= r_steps[n]) break;
402 } while (--n > 0);
403 x0 = this->Ei + n * this->E_step;
404 x1 = x0 + this->E_step;
405 R E_new = mqi::intpl1d(r, r_steps[n], r_steps[n + 1], x0, x1);
406 R dE_mean = rel.Ek - E_new;
407
408 // Add energy straggling (fluctuations around the mean energy loss)
409 R dE_var = this->energy_straggling(rel, mat, length_in_water);
410 R ret = mqi::mqi_normal(rng, dE_mean, mqi::mqi_sqrt(dE_var));
411 if (ret < 0) ret *= -1.0;
412 return ret;
413 }

424 CUDA_HOST_DEVICE
425 virtual void
426 along_step(track_t<R>& trk,
427 track_stack_t<R>& stk,
428 mqi_rng* rng,
429 const R len,
430 material_t<R>& mat) {
431 mqi::relativistic_quantities<R> rel(trk.vtx0.ke, this->units.Mp);
432
433 // 1. Calculate continuous energy loss over the step length
434 R dE = this->energy_loss(rel, mat, len, rng);
435
436 // 2. Check if the particle stops. If so, adjust the step length and energy loss.
437 R r = 1.0;
438 if (dE >= trk.vtx0.ke) {
439 r = trk.vtx0.ke / dE; // scale factor for the step
440 trk.stop();
441 }
442 assert(dE * r >= 0);
443
444 // 3. Calculate Multiple Coulomb Scattering (MCS) angle
445 R P = rel.momentum();
446 R radiation_length_mat = this->radiation_length(mat.rho_mass);
447
448 // Highland's formula for the variance of the projected scattering angle distribution
449 R th_sq = ((this->Es / P) * (this->Es / P) / rel.beta_sq) * len / radiation_length_mat;
450 // Sample the polar scattering angle from a Gaussian distribution
451 R th = mqi::mqi_normal<R>(rng, 0, mqi::mqi_sqrt(th_sq));
452 // Sample the azimuthal angle uniformly
453 R phi = 2.0 * M_PI * mqi::mqi_uniform<R>(rng);
454
455// 4. Update the track's state
456#if !defined(__CUDACC__)
457 if (std::isnan(th) || std::isnan(phi) || std::isinf(th) || std::isinf(phi))
458 printf("p ion1 dE %f ke %f Es %f P %f len %f th_sq %f th %f phi %f\n",
459 dE,
460 trk.vtx0.ke,
461 this->Es,
462 P,
463 len,
464 th_sq,
465 th,
466 phi);
467 if (std::isnan(r)) printf("p ion1 r %f\n", r);
468#endif
469 trk.update_post_vertex_direction(th, phi);
470 if (mqi::mqi_abs(trk.vtx0.dir.dot(trk.vtx1.dir) /
471 (trk.vtx0.dir.norm() * trk.vtx1.dir.norm()) -
472 mqi::mqi_cos(th)) < 1e-3) {
473 } else {
474#ifdef DEBUG
475 printf("cos(th) %f dot %f\n",
476 mqi::mqi_cos(th),
477 trk.vtx0.dir.dot(trk.vtx1.dir) / (trk.vtx0.dir.norm() * trk.vtx1.dir.norm()));
478 printf("vtx0 ");
479 trk.vtx0.dir.dump();
480 printf("vtx1 ");
481 trk.vtx1.dir.dump();
482#endif
483 }
484 assert(mqi::mqi_abs(trk.vtx0.dir.dot(trk.vtx1.dir) /
485 (trk.vtx0.dir.norm() * trk.vtx1.dir.norm()) -
486 mqi::mqi_cos(th)) < 1e-3);
487 trk.deposit(dE * r); // Record the energy deposited in this step
488 trk.update_post_vertex_position(r * len);
489 trk.update_post_vertex_energy(dE * r);
490 }

506 CUDA_HOST_DEVICE
507 virtual void
508 post_step(track_t<R>& trk,
509 track_stack_t<R>& stk,
510 mqi_rng* rng,
511 const R len,
512 material_t<R>& mat,
513 bool score_local_deposit) {
514 mqi::relativistic_quantities<R> rel(trk.vtx1.ke, this->units.Mp);
515
516 // Sample the kinetic energy of a delta electron
517 R Te;
518
519 // Sampling and Rejection from Geant4 to get the correct energy distribution
520 while (true) {
521 Te = sample_delta_energy(rel.Te_max, rng);
522 if (mqi::mqi_uniform<R>(rng) <
523 1.0 - rel.beta_sq * Te / rel.Te_max + Te * Te / (2.0 * rel.Et_sq)) {
524 break;
525 }
526 }
527 assert(Te >= 0);
528
529 // The energy of the created delta electron is assumed to be absorbed locally.
530 // The primary proton loses this amount of energy.
531#ifdef __PHYSICS_DEBUG__
532 // A full simulation might create a new electron track here instead.
533 track_t<R> daughter(trk);
534 daughter.dE = Te;
535 daughter.primary = false;
536 daughter.process = mqi::D_ION;
537 daughter.vtx0.ke = 0;
538 daughter.vtx1.ke = 0;
539 daughter.status = CREATED;
540 daughter.vtx0.pos = trk.c_node->geo->rotation_matrix_fwd *
541 (daughter.vtx0.pos - trk.c_node->geo->translation_vector) +
542 trk.c_node->geo->translation_vector;
543 daughter.vtx0.dir = trk.c_node->geo->rotation_matrix_fwd * daughter.vtx0.dir;
544 daughter.vtx1.pos = trk.c_node->geo->rotation_matrix_fwd *
545 (daughter.vtx1.pos - trk.c_node->geo->translation_vector) +
546 trk.c_node->geo->translation_vector;
547 daughter.vtx1.dir = trk.c_node->geo->rotation_matrix_fwd * daughter.vtx1.dir;
548 stk.push_secondary(daughter);
549#else
550 trk.deposit(Te);
551#endif
552 trk.update_post_vertex_energy(Te);
553 }

563 CUDA_HOST_DEVICE
564 virtual void
565 last_step(track_t<R>& trk, material_t<R>& mat) {
566 mqi::relativistic_quantities<R> rel(trk.vtx0.ke, this->units.Mp);
567 R length_in_water = 0;
568 if (trk.dE > 0) {
569 // Estimate remaining path length based on remaining energy and stopping power
570 length_in_water = -trk.dE / this->dEdx(rel, mat);
571 }
572 // Convert water-equivalent length back to physical length in the current material
573 R step_length = length_in_water * this->units.water_density /
574 (mat.stopping_power_ratio(trk.vtx0.ke) * mat.rho_mass);
575 trk.update_post_vertex_position(step_length);
576 }
