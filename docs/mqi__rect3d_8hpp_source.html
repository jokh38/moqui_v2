240 virtual T
241 operator()(const mqi::vec3<R>& pos) {
242 std::array<size_t, 3> c000_idx = this->find_c000_index(pos);
243 const std::array<R, 6> cell_pts = this->cell_position(c000_idx);
244 const std::array<T, 8> coner = this->cell_data(c000_idx);
245
246 R xd = (pos.x - cell_pts[0]) / (cell_pts[1] - cell_pts[0]);
247 R yd = (pos.y - cell_pts[2]) / (cell_pts[3] - cell_pts[2]);
248 R zd = (pos.z - cell_pts[4]) / (cell_pts[5] - cell_pts[4]);
249
250 T c00 = coner[c000] * (1.0 - xd) + coner[c100] * xd;
251 T c10 = coner[c010] * (1.0 - xd) + coner[c110] * xd;
252
253 T c01 = coner[c001] * (1.0 - xd) + coner[c101] * xd;
254 T c11 = coner[c011] * (1.0 - xd) + coner[c111] * xd;
255
256 T c0 = c00 * (1.0 - yd) + c10 * yd;
257 T c1 = c01 * (1.0 - yd) + c11 * yd;
258
259 return c0 * (1.0 - zd) + c1 * zd;
260 }

626 flip_data(void) {
627 if (flip_[0] == false && flip_[1] == false && flip_[2] == false) { return; }
628
629 std::valarray<T> tmp0(data_.size()); //temporal copy object
630 tmp0 = data_;
631 long int id_from = 0;
632 long int id_to = 0;
633
634 long int idx = 0;
635 long int idy = 0;
636 long int idz = 0;
637
638 for (int k = 0; k < dim_.z; ++k) {
639 for (int j = 0; j < dim_.y; ++j) {
640 for (int i = 0; i < dim_.x; ++i) {
641 idx = (flip_[0]) ? (dim_.x - 1 - i) : i;
642 idy = (flip_[1]) ? (dim_.y - 1 - j) * dim_.x : j * dim_.x;
643 idz = (flip_[2]) ? (dim_.z - 1 - k) * dim_.x * dim_.y : k * dim_.x * dim_.y;
644
645 id_to = this->ijk2data(i, j, k);
646 id_from = idz + idy + idx;
647 data_[id_to] = tmp0[id_from];
648 } //x
649 } //y
