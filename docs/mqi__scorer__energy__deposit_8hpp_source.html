91template<typename R>
92CUDA_DEVICE double
93dose_to_water(const track_t<R>& trk, const cnb_t& cnb, grid3d<mqi::density_t, R>& geo) {
94 R density;
95#if defined(__CUDACC__)
96 density = geo.get_data()[cnb];
97#else
98 density = geo.get_data()[cnb];
99#endif
100 R volume = geo.get_volume(cnb);
101 mqi::h2o_t<R> water;
102 if (density < 1.0e-7) {
103 return 0.0;
104 } else {
105 water.rho_mass = density;
106 // The formula converts energy from MeV to Joules (1.602e-13) and divides by the
107 // mass of the voxel (volume * density) to get dose. The stopping power ratio
108 // adjusts this value to what it would be in water.
109 return (trk.dE + trk.local_dE) * 1.60218e-10 /
110 (volume * density * water.stopping_power_ratio(trk.vtx0.ke));
111 }
112}

125template<typename R>
126CUDA_DEVICE double
127dose_to_medium(const track_t<R>& trk, const cnb_t& cnb, grid3d<mqi::density_t, R>& geo) {
128 R density;
129 density = geo.get_data()[cnb];
130 R volume = geo.get_volume(cnb);
131 // Convert deposited energy (MeV) to Joules (1.602e-13) and divide by mass (kg) to get dose in Gy.
132 return trk.primary ? trk.dE * 1.60218e-13 * 1000.0 / (volume * density)
133 : 0.0;
134}

150template<typename R>
151CUDA_DEVICE double
152LETd_weight1(const track_t<R>& trk, const cnb_t& cnb, grid3d<mqi::density_t, R>& geo) {
153 R density;
154 density = geo.get_data()[cnb];
155 density *= 1000.0; // Convert g/cm^3 to kg/m^3? Or other unit conversion.
156 // Calculate the straight-line distance of the particle's step.
157 double length = (trk.vtx1.pos.x - trk.vtx0.pos.x) * (trk.vtx1.pos.x - trk.vtx0.pos.x);
158 length += (trk.vtx1.pos.y - trk.vtx0.pos.y) * (trk.vtx1.pos.y - trk.vtx0.pos.y);
159 length += (trk.vtx1.pos.z - trk.vtx0.pos.z) * (trk.vtx1.pos.z - trk.vtx0.pos.z);
160 length = mqi::mqi_sqrt(length);
161 if (length <= 0) { return 0.0; }
162 // LET = (Energy deposited) / (path length) / (density)
163 double let = trk.dE / length / density;
164 if (let >= 25.0) { // High LET cut-off
165 return 0;
166 } else {
167 return trk.dE * let;
168 }
169}
