75insert_hashtable(mqi::key_value* hashtable,
76 mqi::key_t key1,
77 mqi::key_t key2,
78 double value,
79 unsigned long long int scorer_offset,
80 uint64_t max_capacity) {
81 mqi::key_t slot;
82 if (value <= 0) { return; }
83 if (key2 == mqi::empty_pair) {
84 slot = key1;
85 key2 = 0;
86 } else {
87 slot = hash_fun(key1, key2, max_capacity);
88 }
89
90 uint32_t prev1, prev2;
91 while (true) {
92#if defined(__CUDACC__)
93 prev1 = atomicCAS(&hashtable[slot].key1, mqi::empty_pair, key1);
94 prev2 = atomicCAS(&hashtable[slot].key2, mqi::empty_pair, key2);
95#else
96 prev1 = CAS(&hashtable[slot].key1, mqi::empty_pair, key1);
97 prev2 = CAS(&hashtable[slot].key2, mqi::empty_pair, key2);
98#endif
99 if ((prev1 == mqi::empty_pair || prev1 == key1) &&
100 (prev2 == mqi::empty_pair || prev2 == key2)) {
101#if defined(__CUDACC__)
102 atomicAdd(&hashtable[slot].value, value);
103#else
104 hashtable[slot].value += value;
105#endif
106 return;
107 }
108 slot = (slot + 1) % (max_capacity);
109 }
