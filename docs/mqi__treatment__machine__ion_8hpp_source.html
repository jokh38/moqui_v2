51 virtual mqi::coordinate_transform<T>
52 create_coordinate_transform(const mqi::dataset* ds, const mqi::modality_type m) {
53 auto seq_tags = &mqi::seqtags_per_modality.at(m);
54 auto layer0 = (*ds)(seq_tags->at("ctrl"))[0]; //layer0
55 std::vector<float> tmp;
56 std::vector<std::string> tmp_dir;
57 std::array<T, 4> angles;
59 layer0->get_values("BeamLimitingDeviceAngle", tmp);
60 angles[0] = tmp[0];
61 layer0->get_values("GantryAngle", tmp);
62 angles[1] = tmp[0];
63
64 layer0->get_values("PatientSupportAngle", tmp);
65 angles[2] = -1.0 * tmp[0];
66 angles[3] = 0.0;
67
68 mqi::vec3<T> pos;
69
71 if ((m == RTPLAN) || (m == IONPLAN)) {
72 layer0->get_values("IsocenterPosition", tmp);
73 pos.x = tmp[0];
74 pos.y = tmp[1];
75 pos.z = tmp[2];
76 }
77 return mqi::coordinate_transform<T>(angles, pos);
