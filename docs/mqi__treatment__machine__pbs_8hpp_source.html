180 virtual mqi::beamlet<T>
181 characterize_beamlet(const mqi::beam_module_ion::spot& s0,
182 const mqi::beam_module_ion::spot& s1) {
183 //energy distribution
184 auto beam_up = beamdata_.lower_bound(s0.e);
185 if (beam_up == beamdata_.begin()) std::cerr << "out-of-bound\n";
186 auto beam_down = std::prev(beam_up, 1);
187 auto down = beam_down->second;
188 auto up = beam_up->second;
189 T mid_e = intpl(s0.e, beam_down->first, beam_up->first, down.E, up.E);
190 T mid_de = intpl(s0.e, down.E, up.E, down.dE, up.dE);
191 T mid_x = intpl(s0.e, down.E, up.E, down.x, up.x);
192 T mid_y = intpl(s0.e, down.E, up.E, down.y, up.y);
193 T mid_xp = intpl(s0.e, down.E, up.E, down.xp, up.xp);
194 T mid_yp = intpl(s0.e, down.E, up.E, down.yp, up.yp);
195 // std::cout<<"Energy: "<< s.e <<", " << down.E << ", " << up.E << ", "<<mid_e<< "\n";
196 auto energy = new mqi::norm_1d<T>({ mid_e }, { mid_de });
197
198 mqi::vec3<T> dir0(std::atan(s0.x / treatment_machine_ion<T>::SAD_[0]),
199 std::atan(s0.y / treatment_machine_ion<T>::SAD_[1]),
200 -1.0);
201 mqi::vec3<T> dir1(std::atan(s1.x / treatment_machine_ion<T>::SAD_[0]),
202 std::atan(s1.y / treatment_machine_ion<T>::SAD_[1]),
203 -1.0);
204
205 mqi::vec3<T> pos0(0, 0, mqi::treatment_machine_ion<T>::source_to_isocenter_mm_);
206 pos0.x = (treatment_machine_ion<T>::SAD_[0] - pos0.z) * dir0.x;
207 pos0.y = (treatment_machine_ion<T>::SAD_[1] - pos0.z) * dir0.y;
208
209 mqi::vec3<T> pos1(0, 0, mqi::treatment_machine_ion<T>::source_to_isocenter_mm_);
210 pos1.x = (treatment_machine_ion<T>::SAD_[0] - pos1.z) * dir1.x;
211 pos1.y = (treatment_machine_ion<T>::SAD_[1] - pos1.z) * dir1.y;
212
213 std::array<T, 6> spot_pos_range = { pos0.x, pos1.x, pos0.y, pos1.y, pos0.z, pos1.z };
214 std::array<T, 6> spot_sigma = { mid_x, mid_y, 0.0, mid_xp, mid_yp, 0.0 };
215 std::array<T, 2> corr = { 0.0, 0.0 };
216
217 //spot direction mean is calculated from the sampled position between x0 and x1
218 //so passing spot-range to a specialized distribution for handling this case is prefered option.
219 auto spot_ = new mqi::phsp_6d_fanbeam<T>(
220 spot_pos_range, spot_sigma, corr, treatment_machine_ion<T>::SAD_);
221
222 return mqi::beamlet<T>(energy, spot_);

337 std::cout << "Thickness is defined from RangeShifter Setting sequence\n";
338 auto rs_ss0 = (*layer0)(seq_tags->at("rsss"))[0];
339
340 //% If MDACC simulates non-water material for rangeshifter,

342 rs_ss0->get_values("RangeShifterWaterEquivalentThickness", ftmp);
344 lxyz.z = ftmp[0] / 1.15; //needs to check later
345

347 rs_ss0->get_values("IsocenterToRangeShifterDistance", ftmp);
348 pxyz.z = ftmp[0];

351
352 } else {
353
354 std::cout << "Thickness is defined from Rangeshifter ID\n";
355 std::vector<float> ftmp;
356 layer0->get_values("SnoutPosition", ftmp);
357 pxyz.z = ftmp[0];
358
359 //3. There must be at least one range shifter sequence
360 for (auto r : rs_ds) {
361 std::vector<std::string> rs_id(0);
362 r->get_values("RangeShifterID", rs_id);
363 auto it = rangeshifter_.find(mqi::trim_copy(rs_id[0]));
364 if (it != rangeshifter_.end()) lxyz.z += it->second;

367 pxyz.z -= (lxyz.z * 0.5 + geometry_spec_.rangeshifter_snout_gap);

382 virtual std::array<T, 2>
383 characterize_beamlet_time(const mqi::beam_module_ion::spot& s_curr,
384 const mqi::beam_module_ion::spot& s_next) {
385 // dT_on: time to beam on
386 // dT_off: time to beam off including layer switing, settling spot, latencies
387 // s_curr.meterset is number of particles
388 const T dT_on = s_curr.meterset * const_e / (time_spec_.i * nA);
389 T dT_off = 0;
390
393 if (&s_next != nullptr) {
394 if (s_next.e != s_curr.e) {
397 dT_off = time_spec_.e * sec;
398 } else {
401 const T dTx = std::abs(s_next.x - s_curr.x) / (time_spec_.vx * m / sec);
402 const T dTy = std::abs(s_next.y - s_curr.y) / (time_spec_.vy * m / sec);
403 if (dTx > 0 || dTy > 0)
404 dT_off =
405 (dTy >= dTx) ? dTy + (time_spec_.set_y * ms) : dTx + (time_spec_.set_x * ms);
406 }
407
409 dT_off += (time_spec_.dt_up * ms);
410 dT_off += (time_spec_.dt_down * ms);
411 ;
412 }
413
414 return { dT_on, dT_off };
415 }
