166 mqi::beamlet<T>
167 characterize_beamlet(const mqi::beam_module_ion::logspot& s,
168 const float source_to_isocenter_mm,
169 const bool rsuse)
170 {
171 // Range shifter correction
172 float newBeamStartingPos{ -source_to_isocenter_mm };
173 if (rsuse) newBeamStartingPos = newBeamStartingPos - 50;
174
175 // Spot beam's energy
176 double energySpread = this->beamEnergySpreadInterp(s.e);
177
178 // Gaussian energy spread distribution
179 auto energy = new mqi::norm_1d<T>({ s.e }, { energySpread });
180
181 // Caculate direction based on SAD and spot's position
182 mqi::vec3<T> dir(std::atan(s.x/treatment_machine_ion<T>::SAD_[0]),
183 std::atan(s.y/treatment_machine_ion<T>::SAD_[1]),
184 -1.0);
185
186 // Determine X,Y position of isocenter
187 mqi::vec3<T> pos(0, 0, -newBeamStartingPos);
188 pos.x = (treatment_machine_ion<T>::SAD_[0] - pos.z) * dir.x ;
189 pos.y = (treatment_machine_ion<T>::SAD_[1] - pos.z) * dir.y ;
190
191 // Spot size interpolation equation (70-230 MeV)
192 double spotSize = this->beamSpotSizeInterp(s.e);
193
194 // Angular spread interpolation equation (70-230 MeV)
195 double angularSpread = this->beamAngularSpreadInterp(s.e);
196
197 // Divergence interpolation equation (70-230 MeV)
198 double divergence = this->beamDivergenceInterp(s.e);
199
200 //Define phsp distribution
201 std::array<T,6> beamlet_mean = { pos.x, pos.y, pos.z, dir.x, dir.y, dir.z };
202 std::array<T,6> beamlet_sigm = { spotSize , spotSize, 0, angularSpread, angularSpread, 0};
203 std::array<T,2> beamlet_divergence = { divergence, divergence };
204 auto beamlet = new mqi::phsp_6d_ray<T>(beamlet_mean, beamlet_sigm, beamlet_divergence, newBeamStartingPos);
205
206 return mqi::beamlet<T>(energy, beamlet);
207 }

318
319 // In single energy layer information

321 {
322 logSpotInfo.muCount = logfileData.beamInfo[i][j].muCount[k];
323 logSpotInfo.x = logfileData.beamInfo[i][j].posX[k];
324 logSpotInfo.y = logfileData.beamInfo[i][j].posY[k];
325 beamsource.append_beamlet_log(this->characterize_beamlet(logSpotInfo, treatment_machine<T>::source_to_isocenter_mm_, rsuse), this->characterize_history(logSpotInfo), pcoord);
326 }
