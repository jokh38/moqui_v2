173 mqi::beamlet<T>
174 characterize_beamlet(const mqi::beam_module_ion::logspot& s,
175 const float source_to_isocenter_mm,
176 const bool rsuse)
177 {
178 // Range shifter correction
179 float newBeamStartingPos{ -source_to_isocenter_mm };
180 if (rsuse) newBeamStartingPos = newBeamStartingPos - 40;
181
182 // Spot beam's energy
183 double energySpread = this->beamEnergySpreadInterp(s.e);
184
185 // Gaussian energy spread distribution
186 auto energy = new mqi::norm_1d<T>({ s.e }, { energySpread });
187
188 // Caculate direction based on SAD and spot's position
189 mqi::vec3<T> dir(std::atan(s.x/treatment_machine_ion<T>::SAD_[0]),
190 std::atan(s.y/treatment_machine_ion<T>::SAD_[1]),
191 -1.0);
192
193 // Determine X,Y position of isocenter
194 mqi::vec3<T> pos(0, 0, -newBeamStartingPos);
195 pos.x = (treatment_machine_ion<T>::SAD_[0] - pos.z) * dir.x ;
196 pos.y = (treatment_machine_ion<T>::SAD_[1] - pos.z) * dir.y ;
197
198 // Spot size interpolation equation (70-230 MeV)
199 double spotSize = this->beamSpotSizeInterp(s.e);
200
201 // Angular spread interpolation equation (70-230 MeV)
202 double angularSpread = this->beamAngularSpreadInterp(s.e);
203
204 // Divergence interpolation equation (70-230 MeV)
205 double divergence = this->beamDivergenceInterp(s.e);
206
207 //Define phsp distribution
208 std::array<T,6> beamlet_mean = { pos.x, pos.y, pos.z, dir.x, dir.y, dir.z };
209 std::array<T,6> beamlet_sigm = { spotSize , spotSize, 0, angularSpread, angularSpread, 0};
210 std::array<T,2> beamlet_divergence = { divergence, divergence };
211 auto beamlet = new mqi::phsp_6d_ray<T>(beamlet_mean, beamlet_sigm, beamlet_divergence, newBeamStartingPos);
212
213 return mqi::beamlet<T>(energy, beamlet);
214 }

307 mqi::beamsource<T>
308 create_beamsource(const mqi::logfiles_t& logfileData,
309 const mqi::coordinate_transform<T> pcoord,
310 const float source_to_isocenter_mm = 465.0,
311 const bool rsuse = false)
312 {
313 // Definition of source to isocenter distance
314 treatment_machine<T>::source_to_isocenter_mm_ = source_to_isocenter_mm;
315
316 // Creating beam source with log file information
317 mqi::beamsource<T> beamsource;
318
319 for (int i = 0; i < logfileData.beamInfo.size(); i++)
320 {
321 for (int j = 0; j < logfileData.beamInfo[i].size(); j++)
322 {
323 mqi::beam_module_ion::logspot logSpotInfo;
324 logSpotInfo.e = logfileData.beamEnergyInfo[i][j]; // Log file spot energy (single energy layer)
325
326 // In single energy layer information
327 for (int k = 0; k < logfileData.beamInfo[i][j].muCount.size(); k++)
328 {
329 logSpotInfo.muCount = logfileData.beamInfo[i][j].muCount[k];
330 logSpotInfo.x = logfileData.beamInfo[i][j].posX[k];
331 logSpotInfo.y = logfileData.beamInfo[i][j].posY[k];
332
333 beamsource.append_beamlet_log(this->characterize_beamlet(logSpotInfo, treatment_machine<T>::source_to_isocenter_mm_, rsuse), this->characterize_history(logSpotInfo), pcoord);
334 }
335 }
336 }
337 return beamsource;
338 }
