54template<typename T, size_t S>
55inline T
56interp_linear(const std::map<T, std::array<T, S>>& db, const T x, const size_t y_col = 0) {
57 auto it_up = db.upper_bound(x);
58 if (it_up == db.end()) { return ((--it_up)->second)[y_col]; }
59 if (it_up == db.begin()) { return (it_up->second)[y_col]; }
60 T x1 = it_up->first;
61 T y1 = (it_up->second)[y_col];
62 auto it_down = --it_up;
63 T x0 = it_down->first;
64 T y0 = (it_down->second)[y_col];
65 return y0 + (x - x0) * (y1 - y0) / (x1 - x0);
66}

106inline float
107TableInterpolation(float* const vector_X,
108 float* const vector_Y,
109 const float x,
110 const int npoints,
111 int order = 4) {
112 float result;
113 // check order of interpolation
114 if (order > npoints) order = npoints;
115 // if x is ouside the vector_X[] interval
116 if (x <= vector_X[0]) return result = vector_Y[0];
117 if (x >= vector_X[npoints - 1]) return result = vector_Y[npoints - 1];
118 // loop to find j so that x[j-1] < x < x[j]
119 int j = 0;
120 while (j < npoints) {
121 if (vector_X[j] >= x) break;
122 j++;
123 }
124 // shift j to correspond to (npoint-1)th interpolation
125 j = j - order / 2;
126 // if j is ouside of the range [0, ... npoints-1]
127 if (j < 0) j = 0;
128 if (j + order > npoints) j = npoints - order;
129 result = 0.0;
130 // Allocate enough space for any table we'd like to read.
131 float* lambda = new float[npoints];
132 for (int is = j; is < j + order; is++) {
133 lambda[is] = 1.0;
134 for (int il = j; il < j + order; il++) {
135 if (il != is)
136 lambda[is] = lambda[is] * (x - vector_X[il]) / (vector_X[is] - vector_X[il]);
137 }
138 result += vector_Y[is] * lambda[is];
139 }
140 delete[] lambda;
141 return result;
142}
