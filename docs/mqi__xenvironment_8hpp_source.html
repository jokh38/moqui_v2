170 reshape_data(int c_ind, int s_ind, mqi::vec3<ijk_t> dim) {
171 double* reshaped_data = new double[dim.x * dim.y * dim.z];
172 int ind_x = 0, ind_y = 0, ind_z = 0, lin = 0;
173 for (int i = 0; i < dim.x * dim.y * dim.z; i++) {
174 reshaped_data[i] = 0;
175 }
176 //printf("max capacity %d\n", this->world->children[c_ind]->scorers[s_ind]->max_capacity_);
177 for (int ind = 0; ind < this->world->children[c_ind]->scorers[s_ind]->max_capacity_;
178 ind++) {
179 if (this->world->children[c_ind]->scorers[s_ind]->data_[ind].key1 != mqi::empty_pair &&
180 this->world->children[c_ind]->scorers[s_ind]->data_[ind].key2 != mqi::empty_pair) {
181 reshaped_data[this->world->children[c_ind]->scorers[s_ind]->data_[ind].key1] +=
182 this->world->children[c_ind]->scorers[s_ind]->data_[ind].value;
183 }
184 }
185 return reshaped_data;
186 }

190 void
191 save_reshaped_files() {
192 //auto start = std::chrono::high_resolution_clock::now();
193 uint32_t vol_size;
194 mqi::vec3<ijk_t> dim;
195 double* reshaped_data;
196 std::string filename;
197 for (int c_ind = 0; c_ind < this->world->n_children; c_ind++) {
198 for (int s_ind = 0; s_ind < this->world->children[c_ind]->n_scorers; s_ind++) {
199 filename =
200 std::to_string(c_ind) + "_" + this->world->children[c_ind]->scorers[s_ind]->name_;
201 dim = this->world->children[c_ind]->geo->get_nxyz();
202 vol_size = dim.x * dim.y * dim.z;
203 reshaped_data = this->reshape_data(c_ind, s_ind, dim);
204 if (!this->output_format.compare("mhd")) {
205 mqi::io::save_to_mhd<R>(this->world->children[c_ind],
206 reshaped_data,
207 1.0,
208 this->output_path,
209 filename,
210 vol_size);
211 } else if (!this->output_format.compare("mha")) {
212 mqi::io::save_to_mha<R>(this->world->children[c_ind],
213 reshaped_data,
214 1.0,
215 this->output_path,
216 filename,
217 vol_size);
218 } else {
219 mqi::io::save_to_bin<double>(
220 reshaped_data, 1.0, this->output_path, filename, vol_size);
221 }
222
223 delete[] reshaped_data;
224 }
225 }
226 //auto stop = std::chrono::high_resolution_clock::now();
227 //std::chrono::duration<double, std::milli> duration = stop - start;
228 //printf("Reshape and save done %f ms\n", duration.count());
229 std::cout << "Simulation result file reshape and saving process complete!" << std::endl;
230 }
