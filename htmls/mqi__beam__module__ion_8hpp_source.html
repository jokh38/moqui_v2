141 beam_module_ion(const mqi::dataset* d, mqi::modality_type m) : beam_module(d, m) {
144 std::vector<int> nb_pts(1);
145 std::vector<float> energy(1);
146 std::vector<float> fwhm_xy(2);
147 std::vector<std::string> tune_id(1);
148 std::vector<float> xy;
149 std::vector<float> weight;
150
151 int layer_nb = 0;
152
153 std::string str_weight;
154
156 switch (modality_) {
157 case mqi::modality_type::IONPLAN:
158 // For plans, weights are the intended meterset values.
159 str_weight = "ScanSpotMetersetWeights";
160 break;
161 case mqi::modality_type::IONRECORD:
162 // For records, weights are the meterset values that were actually delivered.
163 str_weight = "ScanSpotMetersetsDelivered";
164 break;
165 default:
166 // If the modality is neither, throw an error. This is a form of error handling.
167 throw std::runtime_error("Wrong ION type");
168 }
169
174 auto seq_tags = &mqi::seqtags_per_modality.at(m);
175 auto ictrl = (*ds_)(seq_tags->at("ctrl"));
176 for (auto b : ictrl) { // Loop through each control point in the sequence.
177
178 // The `(layer_nb++) % 2` logic skips every other entry (the odd-numbered ones).
179 if ((layer_nb++) % 2) continue;
180
181 // Extract data for the current layer using DICOM tags (string keys).
182 b->get_values("ScanSpotTuneID", tune_id);
183 b->get_values("NominalBeamEnergy", energy);
184 b->get_values("NumberOfScanSpotPositions", nb_pts);
185 b->get_values("ScanningSpotSize", fwhm_xy);
186 b->get_values("ScanSpotPositionMap", xy);
187 b->get_values(str_weight.c_str(), weight);
188
189 // Loop through each spot in the current energy layer.
190 for (int j = 0; j < nb_pts[0]; ++j) {
191 tune_id_ = tune_id[0];
192 // Create a `spot` struct and add it to the `sequence_` vector.
193 sequence_.push_back(
194 { energy[0], xy[j * 2], xy[j * 2 + 1], fwhm_xy[0], fwhm_xy[1], weight[j] });
195 } //per spot
196 // Record the number of spots found in this layer.
197 nb_spots_per_layer_.push_back(nb_pts[0]);
198 } //per layer
199 }
