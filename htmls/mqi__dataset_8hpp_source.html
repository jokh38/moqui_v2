172 dataset(const gdcm::DataSet& d, const bool include_sq = true) : gdcm_ds_(d) {
173 // Get a reference to the global DICOM dictionary for tag lookups.
174 const gdcm::Dicts& dicts = gdcm::Global::GetInstance().GetDicts();
175
176 if (!include_sq) { return; }
177
178 // Iterate over all data elements in the GDCM dataset.
179 for (auto el = gdcm_ds_.Begin(); el != gdcm_ds_.End(); ++el) {
180 const gdcm::Tag& tag = el->GetTag();
181 const gdcm::DictEntry& entry = dicts.GetDictEntry(tag);
182
183 // If the data element is not a sequence (VR != SQ), skip it.
184 if (!(entry.GetVR() & gdcm::VR::SQ)) continue;
185
186 // It is a sequence. Get the sequence of items.
187 // `gdcm::SmartPointer` is a "smart pointer" that manages memory automatically,
188 // similar to how Python handles object references.
189 gdcm::SmartPointer<gdcm::SequenceOfItems> sqi = el->GetValueAsSQ();
190 std::vector<const dataset*> tmp(0);
191 // Loop through each item in the sequence.
192 for (size_t i = 1; i <= sqi->GetNumberOfItems(); ++i) {
193 const gdcm::Item& itm = sqi->GetItem(i);
194 // Recursively create a new mqi::dataset for the nested dataset in the item.
195 // `new` allocates memory, which must be freed later in the destructor.
196 tmp.push_back(new mqi::dataset(itm.GetNestedDataSet()));
197 }
198 // Store the newly created vector of datasets in our lookup table.
199 ds_lut_.push_back(std::make_tuple(tag, entry.GetKeyword(), tmp));
200 }
201 }

315 void
316 get_values(const gdcm::ByteValue* bv,
317 const gdcm::VR vr,
318 const gdcm::VM vm,
319 const gdcm::VL vl,
320 std::vector<float>& res) const {
321 res.clear();
322
323 if (vr & gdcm::VR::VRBINARY) {
324 // For binary representations like FL (Float), the data is stored in native
325 // float format. We can copy the bytes directly.
326 assert(vr & gdcm::VR::FL);
327 size_t ndim = vl / sizeof(float);
328 res.resize(ndim);
329 bv->GetBuffer((char*) &res[0], ndim * sizeof(float));
330 } else if (vr & gdcm::VR::VRASCII) {
331 // For string representations like DS (Decimal String), the numbers are stored
332 // as text. Values are often separated by backslashes ('\').
333 std::string s = std::string(bv->GetPointer(), bv->GetLength());
334 size_t beg = 0;
335 size_t next = std::string::npos;
336 const std::string tok("\\");
337 do {
338 next = s.find_first_of(tok, beg);
339 res.push_back(std::stof(s.substr(beg, next))); // Convert substring to float
340 beg = next + tok.size();
341 } while (next != std::string::npos);
342 }
343 }

352 void
353 get_values(const gdcm::ByteValue* bv,
354 const gdcm::VR vr,
355 const gdcm::VM vm,
356 const gdcm::VL vl,
357 std::vector<int>& res,
358 bool show = false) const {
359 res.clear();
360 size_t ndim = vm.GetLength();
361 if (ndim == 0) {
362 if (vr & gdcm::VR::FL) { // Should be an integer type, but defensive coding
363 ndim = vl / sizeof(int);
364 }
365 }
366 assert(ndim >= 1);
367
368 if (vr & gdcm::VR::VRBINARY) {
369 // For binary integer types (like US, UL), copy the bytes directly.
370 res.resize(ndim);
371 bv->GetBuffer((char*) &res[0], ndim * sizeof(int));
372 } else if (vr & gdcm::VR::VRASCII) {
373 // For string representations like IS (Integer String), parse the string.
374 std::string s = std::string(bv->GetPointer(), bv->GetLength());
375 size_t beg = 0;
376 size_t next = std::string::npos;
377 const std::string tok("\\");
378 do {
379 next = s.find_first_of(tok, beg);
380 res.push_back(std::stoi(s.substr(beg, next))); // Convert substring to integer
381 beg = next + tok.size();
382 } while (next != std::string::npos);
383 }
