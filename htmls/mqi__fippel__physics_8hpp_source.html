130 virtual void
131 stepping(track_t<R>& trk,
132 track_stack_t<R>& stk,
133 mqi_rng* rng,
134 const R& rho_mass,
135 material_t<R>& mat,
136 const R& distance_to_boundary,
137 bool score_local_deposit) {
138
139 // If particle energy is below the transport cutoff, deposit remaining energy and stop.
140 if (trk.vtx0.ke < this->Tp_cut) {
141 if (trk.vtx0.ke < 0) trk.vtx0.ke = 0;
142 assert(trk.vtx0.ke >= 0);
143 trk.deposit(trk.vtx0.ke);
144 trk.update_post_vertex_energy(trk.vtx0.ke);
145 p_ion.last_step(trk, mat);
146 trk.stop();
147 return;
148 }
149
150 mqi::relativistic_quantities<R> rel(trk.vtx0.ke, units.Mp);
151 R length = 0.0;
152 // Determine the maximum allowed step size based on energy loss and geometric constraints.
153 R max_loss_step = max_energy_loss * -1.0 * rel.Ek / p_ion.dEdx(rel, mat);
154 R current_min_step = this->max_step;
155 current_min_step = current_min_step * mat.stopping_power_ratio(rel.Ek) * mat.rho_mass / this->units.water_density;
156 current_min_step = (current_min_step <= max_loss_step) ? current_min_step : max_loss_step;
157 R max_loss_energy = -1.0 * current_min_step * p_ion.dEdx(rel, mat);
158
159 // Calculate the total cross-section (interaction probability) at the current energy
160 // and at the energy after the maximum possible energy loss.
161 R cs1[4] = { p_ion.cross_section(rel, mat),
162 pp_e.cross_section(rel, mat),
163 po_e.cross_section(rel, mat),
164 po_i.cross_section(rel, mat) };
165 R cs1_sum = cs1[0] + cs1[1] + cs1[2] + cs1[3];
166
167 mqi::relativistic_quantities<R> rel_de(trk.vtx0.ke - max_loss_energy, units.Mp);
168 R cs2[4] = { p_ion.cross_section(rel_de, mat),
169 pp_e.cross_section(rel_de, mat),
170 po_e.cross_section(rel_de, mat),
171 po_i.cross_section(rel_de, mat) };
172 R cs2_sum = cs2[0] + cs2[1] + cs2[2] + cs2[3];
173
174 // Use the larger of the two cross-sections for a conservative estimate.
175 R cs_sum = (cs1_sum >= cs2_sum) ? cs1_sum : cs2_sum;
176 R* cs = (cs1_sum >= cs2_sum) ? cs1 : cs2;
177
178 // Sample the Mean Free Path (MFP) - the distance to the next discrete interaction.
179 R prob = mqi_uniform<R>(rng);
180 R mfp = -1.0f * logf(prob) / cs_sum;
181 R step_limit = current_min_step * this->units.water_density / (mat.stopping_power_ratio(rel.Ek) * mat.rho_mass);
182
183 // Branching logic to determine the final step length.
184 // Case 1: Step is limited by the distance to the next voxel boundary.
185 if (distance_to_boundary < mfp && distance_to_boundary < step_limit) {
186 // Take a continuous step to the boundary.
187 p_ion.along_step(trk, stk, rng, distance_to_boundary, mat);
188 assert_track<R>(trk, 0);
189 // Case 2: Step is limited by the Mean Free Path (a discrete interaction occurs).
190 } else if ((mfp < distance_to_boundary ||
191 mqi::mqi_abs(mfp - distance_to_boundary) < mqi::geometry_tolerance) &&
192 (mfp < step_limit || mqi::mqi_abs(mfp - step_limit) < mqi::geometry_tolerance)) {
193 // Take a continuous step for the length of the MFP.
194 p_ion.along_step(trk, stk, rng, mfp, mat);
195 assert_track<R>(trk, 10);
196 if (trk.vtx1.ke < this->Tp_cut) { return; }
197
198 // Sample which discrete interaction occurs.
199 R u = cs_sum * mqi_uniform<R>(rng);
200 trk.vtx1.dir = trk.vtx0.dir;
201 if (u < cs[0]) { // Ionization
202 p_ion.post_step(trk, stk, rng, mfp, mat, score_local_deposit);
203 assert_track<R>(trk, 1);
204 } else if (u < (cs[0] + cs[1])) { // p-p elastic
205 pp_e.post_step(trk, stk, rng, mfp, mat, score_local_deposit);
206 assert_track<R>(trk, 2);
207 } else if (u < (cs[0] + cs[1] + cs[2])) { // p-o elastic
208 po_e.post_step(trk, stk, rng, mfp, mat, score_local_deposit);
209 assert_track<R>(trk, 3);
210 } else if (u < (cs[0] + cs[1] + cs[2] + cs[3])) { // p-o inelastic
211 po_i.post_step(trk, stk, rng, mfp, mat, score_local_deposit);
212 assert_track<R>(trk, 4);
213 }
214 // Case 3: Step is limited by the maximum allowed step size.
215 } else {
216 // Take a continuous step.
217 p_ion.along_step(trk, stk, rng, step_limit, mat);
218 assert_track<R>(trk, 5);
219 }
220 return;
221 }
