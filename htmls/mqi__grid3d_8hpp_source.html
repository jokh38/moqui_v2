566 CUDA_HOST_DEVICE
567 intersect_t<R>
568 intersect(mqi::vec3<R>& p, mqi::vec3<R>& d, mqi::vec3<ijk_t>& idx) {
569 mqi::intersect_t<R> its;
570 its.cell = idx;
571 its.side = mqi::NONE_XYZ_PLANE;
572 its.dist = -1.0;
573
574 R t_max_x = (d.x > 0) ? (xe_[idx.x + 1] - p.x) / d.x : (xe_[idx.x] - p.x) / d.x;
575 R t_max_y = (d.y > 0) ? (ye_[idx.y + 1] - p.y) / d.y : (ye_[idx.y] - p.y) / d.y;
576 R t_max_z = (d.z > 0) ? (ze_[idx.z + 1] - p.z) / d.z : (ze_[idx.z] - p.z) / d.z;
577
578 R min_dist = mqi::p_inf;
579 if (t_max_x > 0 && t_max_x < min_dist) min_dist = t_max_x;
580 if (t_max_y > 0 && t_max_y < min_dist) min_dist = t_max_y;
581 if (t_max_z > 0 && t_max_z < min_dist) min_dist = t_max_z;
582
583 if (min_dist < mqi::p_inf) { its.dist = min_dist; }
584 return its;
585 }

591 CUDA_HOST_DEVICE
592 intersect_t<R>
593 intersect(mqi::vec3<R>& p, mqi::vec3<R>& d) {
594 mqi::intersect_t<R> its;
595 its.dist = -1.0;
596 its.side = mqi::NONE_XYZ_PLANE;
597 its.cell.x = -1;
598 its.cell.y = -1;
599 its.cell.z = -1;
600
601 if (p.x >= xe_[0] && p.x <= xe_[dim_.x] && p.y >= ye_[0] && p.y <= ye_[dim_.y] &&
602 p.z >= ze_[0] && p.z <= ze_[dim_.z]) {
603 its.cell = this->index(p);
604 its.dist = 0;
605 return its;
606 }
607
608 R t_min_x = (xe_[0] - p.x) / d.x;
609 R t_max_x = (xe_[dim_.x] - p.x) / d.x;
610 if (t_min_x > t_max_x) std::swap(t_min_x, t_max_x);
611
612 R t_min_y = (ye_[0] - p.y) / d.y;
613 R t_max_y = (ye_[dim_.y] - p.y) / d.y;
614 if (t_min_y > t_max_y) std::swap(t_min_y, t_max_y);
615
616 if ((t_min_x > t_max_y) || (t_min_y > t_max_x)) return its;
617
618 R t_min = std::max(t_min_x, t_min_y);
619 R t_max = std::min(t_max_x, t_max_y);
620
621 R t_min_z = (ze_[0] - p.z) / d.z;
622 R t_max_z = (ze_[dim_.z] - p.z) / d.z;
623 if (t_min_z > t_max_z) std::swap(t_min_z, t_max_z);
624
625 if ((t_min > t_max_z) || (t_min_z > t_max)) return its;
626
627 t_min = std::max(t_min, t_min_z);
628 t_max = std::min(t_max, t_max_z);
629
630 if (t_min > 0) {
631 its.dist = t_min;
632 mqi::vec3<R> p_on = p + d * its.dist;
633 its.cell = this->index(p_on);
634 }
635
636 return its;
637 }

647 inline mqi::vec3<ijk_t>
648 index(const mqi::vec3<R>& p) {
649 mqi::vec3<ijk_t> idx;
650 idx.x = std::lower_bound(xe_, xe_ + dim_.x + 1, p.x) - xe_ - 1;
651 idx.y = std::lower_bound(ye_, ye_ + dim_.y + 1, p.y) - ye_ - 1;
652 idx.z = std::lower_bound(ze_, ze_ + dim_.z + 1, p.z) - ze_ - 1;
653
654 if (p.x >= xe_[dim_.x]) idx.x = dim_.x - 1;
655 if (p.y >= ye_[dim_.y]) idx.y = dim_.y - 1;
656 if (p.z >= ze_[dim_.z]) idx.z = dim_.z - 1;
657
658 return idx;
659 }
