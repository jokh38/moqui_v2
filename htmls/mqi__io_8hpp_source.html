315template<typename R>
316void
317mqi::io::save_to_npz(const mqi::scorer<R>* src,
318 const R scale,
319 const std::string& filepath,
320 const std::string& filename,
321 mqi::vec3<mqi::ijk_t> dim,
322 uint32_t num_spots) {
323 uint32_t vol_size = dim.x * dim.y * dim.z;
324 const std::string name_a = "indices.npy", name_b = "indptr.npy", name_c = "shape.npy",
325 name_d = "data.npy", name_e = "format.npy";
326
327 // Create temporary vectors to group voxel indices and values by spot index.
328 auto* value_vec = new std::vector<double>[num_spots];
329 auto* vox_vec = new std::vector<mqi::key_t>[num_spots];
330
331 // Iterate through the hash table and populate the temporary vectors.
332 for (int ind = 0; ind < src->max_capacity_; ind++) {
333 if (src->data_[ind].key1 != mqi::empty_pair && src->data_[ind].key2 != mqi::empty_pair) {
334 mqi::key_t vox_ind = src->data_[ind].key1;
335 mqi::key_t spot_ind = src->data_[ind].key2;
336 assert(vox_ind >= 0 && vox_ind < vol_size);
337 value_vec[spot_ind].push_back(src->data_[ind].value * scale);
338 vox_vec[spot_ind].push_back(vox_ind);
339 }
340 }
341
342 // --- Convert the grouped data into the three CSR arrays: data, indices, and indptr ---
343 std::vector<double> data_vec;
344 std::vector<uint32_t> indices_vec;
345 std::vector<uint32_t> indptr_vec;
346
347 int nnz_count = 0;
348 indptr_vec.push_back(nnz_count); // The first entry of indptr is always 0.
349 for (int i = 0; i < num_spots; i++) {
350 data_vec.insert(data_vec.end(), value_vec[i].begin(), value_vec[i].end());
351 indices_vec.insert(indices_vec.end(), vox_vec[i].begin(), vox_vec[i].end());
352 nnz_count += vox_vec[i].size();
353 indptr_vec.push_back(nnz_count);
354 }
355
356 // --- Prepare data for saving with the cnpy library ---
357 uint32_t shape[2] = { num_spots, vol_size };
358 std::string format = "csr";
359
360 // Save the five arrays that constitute a SciPy CSR matrix to a single .npz file.
361 mqi::io::save_npz(filepath + "/" + filename + ".npz", "indices", indices_vec, "w");
362 mqi::io::save_npz(filepath + "/" + filename + ".npz", "indptr", indptr_vec, "a");
363 mqi::io::save_npz(filepath + "/" + filename + ".npz", "shape", shape, 2, "a");
364 mqi::io::save_npz(filepath + "/" + filename + ".npz", "data", data_vec, "a");
365 mqi::io::save_npz(filepath + "/" + filename + ".npz", "format", format, 3, "a");
366
367 delete[] value_vec;
368 delete[] vox_vec;
369}

371template<typename R>
372void
373mqi::io::save_to_npz2(const mqi::scorer<R>* src,
374 const R scale,
375 const std::string& filepath,
376 const std::string& filename,
377 mqi::vec3<mqi::ijk_t> dim,
378 uint32_t num_spots) {
379 // In this voxel-major version, the number of rows is the number of active voxels in the ROI.
380 uint32_t vol_size = src->roi_->get_mask_size();
381 const std::string name_a = "indices.npy", name_b = "indptr.npy", name_c = "shape.npy",
382 name_d = "data.npy", name_e = "format.npy";
383
384 // Create temporary vectors to group spot indices and values by voxel index.
385 auto* value_vec = new std::vector<double>[vol_size];
386 auto* spot_vec = new std::vector<mqi::key_t>[vol_size];
387
388 // Iterate through the hash table and populate the temporary vectors.
389 for (int ind = 0; ind < src->max_capacity_; ind++) {
390 if (src->data_[ind].key1 != mqi::empty_pair && src->data_[ind].key2 != mqi::empty_pair) {
391 // Map the original voxel index to the compact ROI index.
392 mqi::key_t vox_ind = src->roi_->get_mask_idx(src->data_[ind].key1);
393 if (vox_ind < 0) continue; // Skip if the voxel is not in the ROI.
394
395 mqi::key_t spot_ind = src->data_[ind].key2;
396 assert(vox_ind >= 0 && vox_ind < vol_size);
397 value_vec[vox_ind].push_back(src->data_[ind].value * scale);
398 spot_vec[vox_ind].push_back(spot_ind);
399 }
400 }
401
402 // Sort the spot indices for each voxel, as required by the CSR format.
403 for (int ind = 0; ind < vol_size; ind++) {
404 if (spot_vec[ind].size() > 1) {
405 std::vector<int> sort_indices(spot_vec[ind].size());
406 std::iota(sort_indices.begin(), sort_indices.end(), 0);
407 sort(sort_indices.begin(), sort_indices.end(), [&](int i, int j) {
408 return spot_vec[ind][i] < spot_vec[ind][j];
409 });
410 // Reorder the spot and value vectors according to the sorted indices.
411 std::vector<double> sorted_value(spot_vec[ind].size());
412 std::vector<mqi::key_t> sorted_spot(spot_vec[ind].size());
413 for (int sorted_ind = 0; sorted_ind < spot_vec[ind].size(); sorted_ind++) {
414 sorted_value[sorted_ind] = value_vec[ind][sort_indices[sorted_ind]];
415 sorted_spot[sorted_ind] = spot_vec[ind][sort_indices[sorted_ind]];
416 }
417 spot_vec[ind] = sorted_spot;
418 value_vec[ind] = sorted_value;
419 }
420 }
421
422 // --- Convert the grouped data into the three CSR arrays ---
423 std::vector<double> data_vec;
424 std::vector<uint32_t> indices_vec;
425 std::vector<uint32_t> indptr_vec;
426 int spot_count = 0;
427 indptr_vec.push_back(spot_count);
428 for (int i = 0; i < vol_size; i++) {
429 data_vec.insert(data_vec.end(), value_vec[i].begin(), value_vec[i].end());
430 indices_vec.insert(indices_vec.end(), spot_vec[i].begin(), spot_vec[i].end());
431 spot_count += spot_vec[i].size();
432 indptr_vec.push_back(spot_count);
433 }
434
435 // --- Prepare data for saving with the cnpy library ---
436 uint32_t shape[2] = { vol_size, num_spots };
437 std::string format = "csr";
438
439 mqi::io::save_npz(filepath + "/" + filename + ".npz", "indices", indices_vec, "w");
440 mqi::io::save_npz(filepath + "/" + filename + ".npz", "indptr", indptr_vec, "a");
441 mqi::io::save_npz(filepath + "/" + filename + ".npz", "shape", shape, 2, "a");
442 mqi::io::save_npz(filepath + "/" + filename + ".npz", "data", data_vec, "a");
443 mqi::io::save_npz(filepath + "/" + filename + ".npz", "format", format, 3, "a");
444
445 delete[] value_vec;
446 delete[] spot_vec;
447}

449template<typename R>
450void
451mqi::io::save_to_npz(const mqi::scorer<R>* src,
452 const R scale,
453 const std::string& filepath,
454 const std::string& filename,
455 mqi::vec3<mqi::ijk_t> dim,
456 uint32_t num_spots,
457 R* time_scale,
458 R threshold) {
459 uint32_t vol_size = dim.x * dim.y * dim.z;
460 const std::string name_a = "indices.npy", name_b = "indptr.npy", name_c = "shape.npy",
461 name_d = "data.npy", name_e = "format.npy";
462
463 auto* value_vec = new std::vector<double>[num_spots];
464 auto* vox_vec = new std::vector<mqi::key_t>[num_spots];
465
466 // Iterate through the hash table, apply scaling and thresholding, and populate temporary vectors.
467 for (int ind = 0; ind < src->max_capacity_; ind++) {
468 if (src->data_[ind].key1 != mqi::empty_pair && src->data_[ind].key2 != mqi::empty_pair) {
469 mqi::key_t vox_ind = src->data_[ind].key1;
470 mqi::key_t spot_ind = src->data_[ind].key2;
471 assert(vox_ind >= 0 && vox_ind < vol_size);
472 double value = src->data_[ind].value;
473 value *= scale;
474 value -= 2 * threshold;
475 if (value < 0) value = 0;
476 value /= time_scale[spot_ind];
477 value_vec[spot_ind].push_back(value);
478 vox_vec[spot_ind].push_back(vox_ind);
479 }
480 }
481
482 // Convert the grouped data into the three CSR arrays.
483 std::vector<double> data_vec;
484 std::vector<uint32_t> indices_vec;
485 std::vector<uint32_t> indptr_vec;
486 int vox_count = 0;
487 indptr_vec.push_back(vox_count);
488 for (int i = 0; i < num_spots; i++) {
489 data_vec.insert(data_vec.end(), value_vec[i].begin(), value_vec[i].end());
490 indices_vec.insert(indices_vec.end(), vox_vec[i].begin(), vox_vec[i].end());
491 vox_count += vox_vec[i].size();
492 indptr_vec.push_back(vox_count);
493 }
494
495 // Prepare and save the arrays to an .npz file.
496 uint32_t shape[2] = { num_spots, vol_size };
497 std::string format = "csr";
498 mqi::io::save_npz(filepath + "/" + filename + ".npz", "indices", indices_vec, "w");
499 mqi::io::save_npz(filepath + "/" + filename + ".npz", "indptr", indptr_vec, "a");
500 mqi::io::save_npz(filepath + "/" + filename + ".npz", "shape", shape, 2, "a");
501 mqi::io::save_npz(filepath + "/" + filename + ".npz", "data", data_vec, "a");
502 mqi::io::save_npz(filepath + "/" + filename + ".npz", "format", format, 3, "a");
503
504 delete[] value_vec;
505 delete[] vox_vec;
506}

508template<typename R>
509void
510mqi::io::save_to_mhd(const mqi::node_t<R>* children,
511 const double* src,
512 const R scale,
513 const std::string& filepath,
514 const std::string& filename,
515 const uint32_t length) {
516 // Extract geometry information from the node structure.
517 float dx = children->geo[0].get_x_edges()[1] - children->geo[0].get_x_edges()[0];
518 float dy = children->geo[0].get_y_edges()[1] - children->geo[0].get_y_edges()[0];
519 float dz = children->geo[0].get_z_edges()[1] - children->geo[0].get_z_edges()[0];
520 float x0 = children->geo[0].get_x_edges()[0] + dx * 0.5;
521 float y0 = children->geo[0].get_y_edges()[0] + dy * 0.5;
522 float z0 = children->geo[0].get_z_edges()[0] + dz * 0.5;
523
524 // Write the .mhd header file with all necessary metadata.
525 std::ofstream fid_header(filepath + "/" + filename + ".mhd", std::ios::out);
526 if (!fid_header) { std::cout << "Cannot open file!" << std::endl; return; }
527 fid_header << "ObjectType = Image\n";
528 fid_header << "NDims = 3\n";
529 fid_header << "BinaryData = True\n";
530 fid_header << "BinaryDataByteOrderMSB = False\n";
531 fid_header << "CompressedData = False\n";
532 fid_header << "TransformMatrix = 1 0 0 0 1 0 0 0 1\n";
533 fid_header << "Offset = " << x0 << " " << y0 << " " << z0 << std::endl;
534 fid_header << "CenterOfRotation = 0 0 0\n";
535 fid_header << "AnatomicOrientation = RAI\n";
536 fid_header << "DimSize = " << children->geo[0].get_nxyz().x << " "

538 fid_header << "ElementType = MET_DOUBLE\n";
539 fid_header << "ElementSpacing = " << dx << " " << dy << " " << dz << "\n";

541 fid_header.close();
542 if (!fid_header.good()) { std::cout << "Error occurred at writing time!" << std::endl; }
543
544 // Scale the data and write it to the separate .raw file.
545 std::valarray<double> dest(src, length);
546 dest *= scale;
547 std::ofstream fid_raw(filepath + "/" + filename + ".raw", std::ios::out | std::ios::binary);
548 if (!fid_raw) { std::cout << "Cannot open file!" << std::endl; return; }
549 fid_raw.write(reinterpret_cast<const char*>(&dest[0]), length * sizeof(double));
550 fid_raw.close();
551 if (!fid_raw.good()) { std::cout << "Error occurred at writing time!" << std::endl; }
552}
