102 CUDA_HOST_DEVICE
103 virtual R
104 radiation_length() {
105 R radiation_length_mat = 0.0;
106 R f = 0.0;
107 R density_tmp = this->rho_mass * 1000.0;
108 if (density_tmp <= 0.26) {
109 f = 0.9857 + 0.0085 * density_tmp;
110 } else if (density_tmp > 0.26 && density_tmp <= 0.9) {
111 f = 1.0446 - 0.2180 * density_tmp;
112 } else if (density_tmp > 0.9) {
113 // f = 1.19 + 0.44 * mqi::mqi_ln(density_tmp - 0.44);
114 f = 1.19 + 0.44 * logf(density_tmp - 0.44);
115 }
116 radiation_length_mat = (this->units.water_density * this->units.radiation_length_water) /
117 (density_tmp * 1e-3 * f);
118 return radiation_length_mat;
119 }
