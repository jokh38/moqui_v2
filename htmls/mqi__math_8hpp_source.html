445template<>
446CUDA_DEVICE float
447mqi_exponential<float, mqi_rng>(mqi_rng* rng, float avg, float up) {
448 float x;
449 do {
450 // Inverse transform sampling for exponential distribution
451 x = -1.0 / avg * logf(1.0 - curand_uniform(rng));
452 } while (x > up || mqi::mqi_isnan(x)); // Reject samples outside the upper bound
453 return x;
454}

638template<>

640mqi_exponential<float, mqi_rng>(mqi_rng* rng, float avg, float up) {
641 float x;
642 std::exponential_distribution<float> dist(1.0 / avg); // C++ uses lambda (1/mean)
643 // The rejection sampling to respect the upper bound seems to be commented out.
644 // This might be an incomplete implementation for the CPU side.
645 x = dist(*rng);
646 return x;
647}

649template<>
650double
651mqi_exponential<double, mqi_rng>(mqi_rng* rng, double avg, double up) {
652 double x;
653 std::exponential_distribution<double> dist(1.0 / avg);
654 do {
655 x = dist(*rng);
656 } while (x > up || x <= 0);
657 return x;
