206 R cs = 0;
207
208 if (rel.Ek >= Ek_min && rel.Ek <= Ek_max) {
209 uint16_t idx0 = uint16_t((rel.Ek - Ek_min) / dEk); //0 - 598
210 uint16_t idx1 = idx0 + 1;
211 R x0 = Ek_min + idx0 * dEk;
212 R x1 = x0 + 0.5;
213 cs = mqi::intpl1d<R>(rel.Ek, x0, x1, cs_table[idx0], cs_table[idx1]);
214 }

251 if (Ek <= 215 && Ek > 200) {

255 power = 0.4;
256 } else if (Ek > 215) {
257 Prob_2nd = 0.78;
258 Prob_long =
259 Prob_2nd + (1 - Prob_2nd) * 1.0;
260 power = 0.4;
261 } else if (Ek <= 200 && Ek > 150) {
262 Prob_2nd = 0.72;
263 Prob_long =
264 Prob_2nd + (1 - Prob_2nd) * 0.83;
265 power = 0.45;

268 Prob_long =

270 power = 0.52;
271 }
272 while ((Er - Eb) > this->E_mini) {
273 Er -= Eb;
275 R u = mqi::mqi_uniform<R>(rng);
276 R dE = mqi::mqi_pow<R>(u, power) * (Er - this->E_mini) + this->E_mini;
277 assert(dE >= 0);
278 if (dE >= Er) dE = Er; //In case deposit energy is greater than E_remain
279 Er -= dE; //substract deposit energy
280
282 assert(dE + Eb >= 0);

284 dE_total += (dE + Eb);
285
288 R zeta = mqi::mqi_uniform<R>(rng);
289 if (zeta < Prob_2nd) {
292 R cos_th = (2.0 * dE / Ek - 1.0) + 2.0 * (1 - dE / Ek) * mqi::mqi_uniform<R>(rng);
293 if (cos_th < -1) cos_th = -1;
294 if (cos_th > 1) cos_th = 1;
295 R th = mqi::mqi_acos<R>(cos_th);
296 R phi = 2.0 * M_PI * mqi::mqi_uniform<R>(rng);
297

299 daughter.dE = 0;
300 daughter.local_dE = 0;

310 (daughter.vtx1.pos - trk.c_node->geo->translation_vector) +
311 trk.c_node->geo->translation_vector;

317 stk.push_secondary(daughter);
318 E_2nd += dE;
319
320 } else if (zeta < Prob_long) {
323 E_long += dE;
324 } else {
