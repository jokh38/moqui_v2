158 void
159 insert_pair(mqi::key_t key1, mqi::key_t key2, R value, unsigned long long int scorer_offset) {
160 mqi::key_t slot;
161 if (key2 == mqi::empty_pair) {
162 slot = key1;
163 key2 = 0;
164 } else {
165 slot = hash_fun(key1 + (key2 * scorer_offset));
166 }
167
168 uint32_t prev1, prev2;
169 while (true) {
170#if defined(__CUDACC__)
171 prev1 = atomicCAS(&this->data_[slot].key1, mqi::empty_pair, key1);
172 prev2 = atomicCAS(&this->data_[slot].key2, mqi::empty_pair, key2);
173#else
174 prev1 = CAS(&this->data_[slot].key1, mqi::empty_pair, key1);
175 prev2 = CAS(&this->data_[slot].key2, mqi::empty_pair, key2);
176#endif
177 if ((prev1 == mqi::empty_pair || prev1 == key1) &&
178 (prev2 == mqi::empty_pair || prev2 == key2)) {
179#if defined(__CUDACC__)
180 atomicAdd(&this->data_[slot].value, value);
181#else
182 this->data_[slot].value += value;
183#endif
184 return;
185 }
186 slot = (slot + 1) % (this->max_capacity_ - 1);
187 }
188 }

201 CUDA_DEVICE
202 virtual void
203 process_hit(const track_t<R>& trk,
204 const int32_t& cnb,
205 grid3d<mqi::density_t, R>& geo,
206 const uint32_t& offset,
207 unsigned long long int scorer_offset = 0) {
208 // Calculate index to store hit
209 // idx : -1 => a hit occured out of ROI. nothing to do.
210 int32_t idx = roi_->idx(cnb);
211 if (idx == -1) return;
212
214 R quantity = (*this->compute_hit_)(trk, cnb, geo);
215
217#if defined(__CUDACC__)
218 insert_pair(cnb, offset, quantity, scorer_offset);
219
220 if (this->score_variance_) {
221 atomicAdd(&count_[cnb].value, 1.0);
222 R delta = quantity - mean_[cnb].value;
223 atomicAdd(&mean_[cnb].value, delta / count_[cnb].value);
224 atomicAdd(&variance_[cnb].value, delta * (quantity - mean_[cnb].value));
225 }
226#else
227 mtx.lock();
228 insert_pair(cnb, offset, quantity, scorer_offset);
229 data_[idx].value += quantity;
230 if (this->score_variance_) {
231 count_[cnb].value += 1.0;
232 R delta = quantity - mean_[cnb].value;
233 mean_[cnb].value += delta / count_[cnb].value;
234 variance_[cnb].value += delta * (quantity - mean_[cnb].value);
235 }
236
237 mtx.unlock();
238#endif
239 }
