131void
132mqi::io::parse_zip_footer(std::string filename,
133 uint16_t& nrecs,
134 size_t& global_header_size,
135 size_t& global_header_offset) {
136 std::ifstream fp(filename.c_str(), std::ios::in | std::ios::binary);
137 std::vector<char> footer(22);
138 fp.seekg(-22, fp.end);
139 fp.read(&footer[0], 22);
140 if (!fp) {
141 std::cout << "error: only " << fp.gcount() << " could be read" << std::endl;
142 exit(-1);
143 }
144 uint16_t disk_no, disk_start, nrecs_on_disk, comment_len;
145 disk_no = *(uint16_t*) &footer[4];
146 disk_start = *(uint16_t*) &footer[6];
147 nrecs_on_disk = *(uint16_t*) &footer[8];
148 nrecs = *(uint16_t*) &footer[10];
149 global_header_size = *(uint32_t*) &footer[12];
150 global_header_offset = *(uint32_t*) &footer[16];
151 comment_len = *(uint16_t*) &footer[20];
152 assert(disk_no == 0);
153 assert(disk_start == 0);
154 assert(nrecs_on_disk == nrecs);
155 assert(comment_len == 0);
156 fp.seekg(global_header_offset, fp.beg);
157
158 fp.close();
159}

198 return 'c';
