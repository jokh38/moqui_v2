215 update_post_vertex_direction(const R& theta, const R& phi) {
216 // Create a local rotation matrix from the scattering angles.
217 mqi::mat3x3<R> m_local(0, theta, phi);
218 // Apply this rotation to a reference vector (0,0,1).
219 mqi::vec3<R> d_local = m_local * ref_vector;
220 d_local.normalize();
221 // Create a global rotation matrix that maps the reference vector to the particle's actual direction.
222 mqi::mat3x3<R> m_global(ref_vector, vtx1.dir);
223 // Apply the global rotation to the locally-rotated vector to get the final new direction.
224 vtx1.dir = m_global * d_local;
225 vtx1.dir.normalize();
226 }
