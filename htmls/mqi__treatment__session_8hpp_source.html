68 treatment_session(std::string file_for_tx_machine, //for beamline and source,
69 std::string m_name = "",
70 std::string mc_code = "TOPAS 3.5",
71 int gantryNum = 2) {
72
73 // Read RT ION PLAN/RECORD with GDCM
74 gdcm::Reader reader;
75 reader.SetFileName(file_for_tx_machine.c_str());
76
77 const bool is_file_valid = reader.Read();
78 if (!is_file_valid)
79 throw std::runtime_error("Invalid DICOM file is given to treatment_session.");
80
81 // Check media storage modality
82 // Declare mtype based on media storage name of DICOM file
83 gdcm::MediaStorage ms;
84 ms.SetFromFile(reader.GetFile());
85
86 switch (ms) {
87 case gdcm::MediaStorage::RTPlanStorage:
88 mtype_ = RTPLAN;
89 break;
90 case gdcm::MediaStorage::RTIonPlanStorage:
91 mtype_ = IONPLAN;
92 break;
93 case gdcm::MediaStorage::RTIonBeamsTreatmentRecordStorage:
94 mtype_ = IONRECORD;
95 break;
96 default:
97 throw std::runtime_error("treatment_session does not supports given RTMODALITY");
98 }
99
100 // Get dataset and modality
101 mqi_ds_ = new mqi::dataset(reader.GetFile().GetDataSet(), true);
102 seq_tags_ = &mqi::seqtags_per_modality.at(mtype_);
103
105 // If there is not beam name in function, use DICOM tag (beam name) in the file.
106 if (!m_name.compare("")) {
109 const mqi::dataset* beam_ds;
110 if (mtype_ == IONPLAN) {
111 beam_ds = (*mqi_ds_)(seq_tags_->at("beam"))[0];
112 } else if (mtype_ == IONRECORD) {
113 beam_ds = (*mqi_ds_)(seq_tags_->at("machine"))[0];
114 } else {
115 throw std::runtime_error(
116 "treatment_session can't find machine name and institute.");
117 }
118
119 std::vector<std::string> in;
120 beam_ds->get_values("InstitutionName", in);
121 std::vector<std::string> bn;
122 beam_ds->get_values("TreatmentMachineName", bn);
123 assert(bn.size() > 0);
124
125 machine_name_ = (in.size() == 0) ? "" : mqi::trim_copy(in[0]);
126 machine_name_ += ":" + mqi::trim_copy(bn[0]);
127
128 } else {
129 machine_name_ = m_name;
130 }
131
132 if (!this->create_machine(machine_name_, mc_code, gantryNum)) {
133 std::runtime_error("No MC machine is registered for " + machine_name_);
134 }
135 }
